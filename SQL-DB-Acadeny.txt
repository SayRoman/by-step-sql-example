CREATE DATABASE Academy;

USE Academy;

 CREATE TABLE Groupp ( Id INT NOT NULL AUTO_INCREMENT, Name NVARCHAR(10) NOT NULL UNIQUE CHECK(Name = ''), Rating INT NOT NULL Check(Rating >=0), CHECK( Rating <6), Year INT NOT NULL CHECK(Year > 0), CHECK(Year <6), PRIMARY KEY(Id));

CREATE TABLE Depatments (Id INT NOT NULL AUTO_INCREMENT, Financing MONEY NOT NULL CHEK(	Financing>=0) DEFAULT '0', Name NVARCHAR(100) NOT NULL CHECK(Name='') UNIQUE, PRIMARY KEY(Id));

CREATE TABLE Faculties (Id INT NOT NULL AUTO_INCREMENT,PRIMARY KEY(Id), Name NVARCHAR(100) NOT NULL UNIQUE CHECK(Name='');

CREATE TABLE Teachers (Id INT NOT NULL AUTO_INCREMENT, EmployementDate DATE NOT NULL, Name NVARCHAR(max) NOT NULL CHECK(Name=''), Premium MONEY NOT NULL CHEK(Premium>=0) DEFAULT '0', Salary MONEY NOT NULL CHEK(Premium>0), Surname NVARCHAR(max) NOT NULL CHECK(Surname =''), PRIMARY KEY(Id));
